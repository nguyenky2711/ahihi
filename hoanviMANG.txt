#include<iostream>
using namespace std;
#define MAX 100
void Nhap_mang(int a[], int &n)
{
	do {
		cout << "\n Nhap so luong phan tu cua mang: ";
		cin >> n;
		if (n<0 || n>MAX)
		{
			cout << "\n Loi, Yeu cau nhap lai!! ";
		}
	} while (n<0 || n>MAX);
	for (int i = 0;i < n;i++)
	{
		cout << "\n Nhap phan tu a[" << i << "]: ";
		cin >> a[i];
	}

}
void Xuat_mang(int a[], int n)
{
	for (int i = 0;i < n;i++)
	{

		cout << a[i] << "   ";
	}
}
//hàm hoán vị các phần tử trong mảng theo chiều tăng dần
void Hoan_vi_HAM_MIN(int a[], int n) 
{
	cout << " \n\n Hoan vi ham Min \n";
	for (int i = 0;i < n;i++)
	{
		for (int j = i+1;j < n;j++) //bắt đầu ở a[i] ( a[0] ) nếu thỏa điều kiện thì đổi chỗ a[j] vs a[i]
		{                           // nếu k thỏa điều kiện thì tiếp tục tăng j cho đến khi xuất
			int c = a[i];
			if (a[j] < a[i])
			{
				c = a[j];
				a[j] = a[i];
				a[i] = c;				
			}
		}
		cout << a[i] << "  ";
	}	
}
int main()
{
	int a[MAX];
	int n;
	Nhap_mang(a, n);
	Xuat_mang(a, n);
	Hoan_vi_HAM_MIN(a, n);
	system("pause");
	return 0;
}
