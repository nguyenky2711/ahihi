#include<iostream>
using namespace std;
#define MAX 100 //khai báo biến hàng MAX =100
void Nhap_Mang(int a[], int n) //hàm nhập mảng
{
	for (int i = 0;i < n;i++) //vòng lặp nhập các phần tử của mảng
	{
		cout << "\nNhap phan tu a[" << i << "] = " ;
		cin >>a[i];
	}
}
void Xuat_Mang(int a[], int n) //hàm xuất giá trị mảng
{
	cout << "Mang: " << endl;
	for(int i=0;i<n;i++) //vòng lặp xuất tất cả các giá trị của mảng
	{
		cout << a[i] << " ";
	}
}
int Tong_Phantu_Mang(int a[], int n) //hàm cộng các phần tử của mảng
{
	int Tong = 0;
	for (int i = 0;i < n;i++) //vòng lập cộng tất cả các giá trị của mảng
	{
		Tong += a[i];
	}
	return Tong;
}
int main()
{
	int a[MAX]; //khai báo mảng số nguyên có 100 phần tử
	int n; // số lượng phần từ của mảng cần dùng
	do
	{
		cout << "\nNhap so luong phan tu mang";
		cin >> n;
		if (n<1 || n>MAX)
		{
			cout << "\nYeu cau nhap lai!!";
			system("pause");
		}
	} while (n<1 || n>MAX);
	cout << "\n\n\t\t NHAP MANG \n";
	Nhap_Mang(a, n);
	cout << "\n\n\t\t XUAT MANG \n";
	Xuat_Mang(a, n);
	cout << "\n\n\t\t TONG CAC PHAN TU \n" << Tong_Phantu_Mang(a, n);
	system("pause");
	return 0;

}
